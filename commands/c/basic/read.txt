template<class T>
inline void read(T& x){
    x = 0; int f = 0; char c = 0;
    while(!isdigit(c)) f |= c == '-', c = getchar();
    while(isdigit(c)) x = x * 10 + c - '0', c = getchar();
    x = f ? -x : x;
}
template<class T1, class... Tn>
inline void read(T1& x1, Tn&... xn){
    read(x1), read(xn...);
}
